void print_line(int n) {
  //n - length
  for (int i = 0; i < n; i++) {
    Serial.print("*");
  }
  Serial.println("");
}
void triangle(int n) {
  //n - height
  for (int i = 0; i < n; i++) {
    print_line(i * 2 + 1);
  }
}
void christmas_tree(int n) {
  //n - height
  for (int i = 3; i < 3+n; i++) {
    triangle(i);
  }
}
